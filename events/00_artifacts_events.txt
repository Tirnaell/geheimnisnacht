character_event = {
	id = ART.0 #MNM.9000
	hide_window = yes
	is_triggered_only = yes
	has_dlc = "Mystics"
	min_age = 13
	only_playable = yes
	only_capable = yes
	prisoner = no

	trigger = {
		any_courtier_or_vassal = {
			NOT = {
				trait = incapable
			}
			prisoner = no
			opinion = {
				who = ROOT
				value = 20
			}
			is_councillor = yes
			ai = yes
		}
		NOR = {
			has_character_modifier = relic_rumor_major
			has_character_modifier = relic_rumor_medium
			has_character_modifier = relic_rumor_minor
		}
	}

	immediate = {
		log = "work event id = ART.0 for [Root.GetBestName]"
		random_courtier_or_vassal = {
			limit = {
				NOT = {
					trait = incapable
				}
				prisoner = no
				opinion = {
					who = ROOT
					value = 20
				}
				is_councillor = yes
				ai = yes
			}
			save_event_target_as = rumor_councillor
		}
		save_event_target_as = relic_ruler
		random_list = {
			90 = {
				#Good rumor
				modifier = {
					factor = 0
					has_artifact = crown_of_command
				}
				event_target:rumor_councillor = {
					character_event = {
						id = ART.2
					}
				}
			}
			100 = {
				#Weak rumor
				modifier = {
					factor = 0
					has_artifact = ring_of_persuasion
					has_artifact = amulet_of_protection_1
					has_artifact = potion_of_healing
					has_artifact = sword_1
					has_artifact = sword_2
				}
				event_target:rumor_councillor = {
					character_event = {
						id = ART.3
					}
				}
			}
			9000 = {
				#AI is less likely to find artifacts
				modifier = {
					factor = 0
					ai = no
				}
			}
		}
	}
}

	has_dlc = "Mystics"

		if = {
			#Stop hunting for relics
			limit = {
				OR = {
					is_alive = no
					NOT = {
						has_character_flag = relic_hunter
					}
					is_councillor = no
					trait = incapable
					prisoner = yes
					liege = {
						NOT = {
							character = event_target:relic_ruler
						}
					}

							has_character_modifier = relic_rumor_major

						}
					}
				}
			}
			if = {
				limit = {
					has_character_flag = relic_hunter
				}
				event_target:relic_ruler = {
					character_event = {
						id = ART.9
					}
				}
			}
			if = {
				limit = {
					NOT = {
						has_character_flag = relic_hunter
					}
				}
				clr_artifact_searcher_effect = yes
			}
			break = yes
		}
		#If relic is found...
		if = {
			limit = {
				OR = {
					AND = {
						event_target:relic_ruler = {
							check_variable = {
								which = relic_score
								value = 15
							}
						}

							has_character_modifier = relic_rumor_minor

					}

						}

			}
