		}

			factor = 1.25

		}
		modifier = {
			factor = 1.5
			trait = poet
		}
		modifier = {
			factor = 1.5
			trait = falconer
		}
		modifier = {
			factor = 1.5
			trait = gregarious
		}
		modifier = {
			factor = 2.0
			trait = ambitious
		}
		modifier = {
			factor = 1.5
			trait = naive_appeaser
		}
		modifier = {
			factor = 2.0
			trait = underhanded_rogue
		}

			factor = 10.0

				trait = charismatic_negotiator
